47  /*
48  * This function implements the central flux as numerical flux function.

50  inline DEAL_II_ALWAYS_INLINE //
51  vector
52  calculate_flux(vector const & value_m, vector const & value_p) const
53  {
54  return 0.5 * (value_m + value_p);

57  /*
58  * Volume flux, i.e., the term occurring in the volume integral for
59  * weak formulation (performing integration-by-parts)
60  */
61  inline DEAL_II_ALWAYS_INLINE //
62  vector
63  get_volume_flux_weak(CellIntegratorU & velocity, unsigned int const q) const
64  {
65  // minus sign due to integration by parts
66  return -velocity.get_value(q);
67  }
68 
69  /*
70  * Volume flux, i.e., the term occurring in the volume integral for
71  * strong formulation (no integration-by-parts, or integration-by-parts performed twice)
72  */
73  inline DEAL_II_ALWAYS_INLINE //
74  scalar
75  get_volume_flux_strong(CellIntegratorU & velocity, unsigned int const q) const
76  {
77  return velocity.get_divergence(q);
