40template<int dim, typename Number>
41class GridMotionElasticity : public GridMotionBase<dim, Number>
42{
43public:
44 typedef LinearAlgebra::distributed::Vector<Number> VectorType;
45
49 GridMotionElasticity(std::shared_ptr<Mapping<dim> const> mapping_undeformed,
50 std::shared_ptr<Structure::Operator<dim, Number>> structure_operator,
51 Structure::Parameters const & structure_parameters)
52 : GridMotionBase<dim, Number>(mapping_undeformed,
53 // extract mapping_degree_moving from elasticity operator
54 structure_operator->get_dof_handler().get_fe().degree,
55 structure_operator->get_dof_handler().get_triangulation()),
56 pde_operator(structure_operator),
57 param(structure_parameters),
58 pcout(std::cout,
59 Utilities::MPI::this_mpi_process(
60 structure_operator->get_dof_handler().get_communicator()) == 0),
61 iterations({0, {0, 0}})
62 {
63 pde_operator->initialize_dof_vector(displacement);
64 }
65
69 void
70 update(double const time, bool const print_solver_info) override
71 {
72 Timer timer;
73 timer.restart();
74
75 if(param.large_deformation) // nonlinear problem
76 {
77 VectorType const_vector;
78
79 auto const iter = pde_operator->solve_nonlinear(
80 displacement, const_vector, 0.0 /* no mass term */, time, param.update_preconditioner);
81
82 iterations.first += 1;
83 std::get<0>(iterations.second) += std::get<0>(iter);
84 std::get<1>(iterations.second) += std::get<1>(iter);
85
86 if(print_solver_info)
87 {
88 this->pcout << std::endl << "Solve moving mesh problem (nonlinear elasticity):";
89 print_solver_info_nonlinear(pcout, std::get<0>(iter), std::get<1>(iter), timer.wall_time());
90 }
91 }
92 else // linear problem
93 {
94 // calculate right-hand side vector
95 VectorType rhs;
96 pde_operator->initialize_dof_vector(rhs);
97 pde_operator->compute_rhs_linear(rhs, time);
98
99 auto const iter = pde_operator->solve_linear(displacement, rhs, 0.0 /* no mass term */, time);
100
101 iterations.first += 1;
102 std::get<1>(iterations.second) += iter;
103
104 if(print_solver_info)
105 {
106 this->pcout << std::endl << "Solve moving mesh problem (linear elasticity):";
107 print_solver_info_linear(pcout, iter, timer.wall_time());
108 }
109 }
110
111 this->moving_mapping->initialize_mapping_q_cache(this->mapping_undeformed,
112 displacement,
113 pde_operator->get_dof_handler());
114 }
115
119 void
120 print_iterations() const override
121 {
122 std::vector<std::string> names;
123 std::vector<double> iterations_avg;
124
125 if(param.large_deformation)
126 {
127 names = {"Nonlinear iterations",
128 "Linear iterations (accumulated)",
129 "Linear iterations (per nonlinear it.)"};
130
131 iterations_avg.resize(3);
132 iterations_avg[0] =
133 (double)std::get<0>(iterations.second) / std::max(1., (double)iterations.first);
134 iterations_avg[1] =
135 (double)std::get<1>(iterations.second) / std::max(1., (double)iterations.first);
136 if(iterations_avg[0] > std::numeric_limits<double>::min())
137 iterations_avg[2] = iterations_avg[1] / iterations_avg[0];
138 else
139 iterations_avg[2] = iterations_avg[1];
140 }
141 else // linear
142 {
143 names = {"Linear iterations"};
144 iterations_avg.resize(1);
145 iterations_avg[0] =
146 (double)std::get<1>(iterations.second) / std::max(1., (double)iterations.first);
147 }
148
149 print_list_of_iterations(pcout, names, iterations_avg);
150 }
151
152private:
153 std::shared_ptr<Structure::Operator<dim, Number>> pde_operator;
154
155 Structure::Parameters const & param;
156
157 // store solution of previous time step / iteration so that a good initial
158 // guess is available in the next step, easing convergence or reducing computational
159 // costs by allowing larger tolerances
160 VectorType displacement;
161
162 ConditionalOStream pcout;
163
164 std::pair<
165 unsigned int /* calls */,
166 std::tuple<unsigned long long, unsigned long long> /* iteration counts {Newton, linear}*/>
167 iterations;
168};
