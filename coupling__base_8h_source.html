168template<int dim, int data_dim, typename VectorizedArrayType>
169CouplingBase<dim, data_dim, VectorizedArrayType>::CouplingBase(
170 std::shared_ptr<dealii::MatrixFree<dim, double, VectorizedArrayType> const> matrix_free_,
171#ifdef EXADG_WITH_PRECICE
172 std::shared_ptr<precice::SolverInterface> precice,
173#endif
174 std::string const mesh_name,
175 dealii::types::boundary_id const surface_id)
176 : matrix_free(matrix_free_),
177#ifdef EXADG_WITH_PRECICE
178 precice(precice),
179#endif
180 mesh_name(mesh_name),
181 dealii_boundary_surface_id(surface_id),
182 write_data_type(WriteDataType::undefined)
183{
184 Assert(matrix_free_.get() != nullptr, dealii::ExcNotInitialized());
185
186#ifdef EXADG_WITH_PRECICE
187 Assert(precice.get() != nullptr, dealii::ExcNotInitialized());
188
189 // Ask preCICE already in the constructor for the IDs
190 mesh_id = precice->getMeshID(mesh_name);
191#else
192 AssertThrow(false,
193 dealii::ExcMessage("EXADG_WITH_PRECICE has to be activated to use this code."));
194 mesh_id = 0;
195#endif
196}

256template<int dim, int data_dim, typename VectorizedArrayType>
257void
258CouplingBase<dim, data_dim, VectorizedArrayType>::print_info(bool const reader,
259 unsigned int const local_size) const
260{
261 Assert(matrix_free.get() != 0, dealii::ExcNotInitialized());
262 dealii::ConditionalOStream pcout(std::cout,
263 dealii::Utilities::MPI::this_mpi_process(
264 matrix_free->get_dof_handler().get_communicator()) == 0);
265 auto const map = (reader ? read_data_map : write_data_map);
266
267 auto names = map.begin();
268 std::string data_names = names->first;
269 ++names;
270 for(; names != map.end(); ++names)
271 data_names += std::string(", ") + names->first;
272
273 pcout << "-- Data " << (reader ? "reading" : "writing") << ":\n"
274 << "-- . data name(s): " << data_names << "\n"
275 << "-- . associated mesh: " << mesh_name << "\n"
276 << "-- . Number of coupling nodes: "
277 << dealii::Utilities::MPI::sum(local_size,
278 matrix_free->get_dof_handler().get_communicator())
279 << "\n"
280 << "-- . Node location: " << get_surface_type() << "\n"
281 << std::endl;
282}
