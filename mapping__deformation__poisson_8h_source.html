40template<int dim, typename Number>
41class DeformedMapping : public DeformedMappingBase<dim, Number>
42{
43public:
44 typedef dealii::LinearAlgebra::distributed::Vector<Number> VectorType;
45
49 DeformedMapping(std::shared_ptr<Grid<dim> const> grid,
50 std::shared_ptr<dealii::Mapping<dim> const> mapping_undeformed,
51 std::shared_ptr<Poisson::BoundaryDescriptor<1, dim> const> boundary_descriptor,
52 std::shared_ptr<Poisson::FieldFunctions<dim> const> field_functions,
53 Poisson::Parameters const & param,
54 std::string const & field,
55 MPI_Comm const & mpi_comm)
56 : DeformedMappingBase<dim, Number>(mapping_undeformed, param.degree, *grid->triangulation),
57 pcout(std::cout, dealii::Utilities::MPI::this_mpi_process(mpi_comm) == 0),
58 iterations({0, 0})
59 {
60 // initialize PDE operator
61 pde_operator = std::make_shared<Poisson::Operator<dim, dim, Number>>(
62 grid, mapping_undeformed, boundary_descriptor, field_functions, param, field, mpi_comm);
63
64 // initialize matrix_free_data
65 matrix_free_data = std::make_shared<MatrixFreeData<dim, Number>>();
66
67 if(param.enable_cell_based_face_loops)
68 Categorization::do_cell_based_loops(*grid->triangulation, matrix_free_data->data);
69
70 matrix_free_data->append(pde_operator);
71
72 // initialize matrix_free
73 matrix_free = std::make_shared<dealii::MatrixFree<dim, Number>>();
74 matrix_free->reinit(*mapping_undeformed,
75 matrix_free_data->get_dof_handler_vector(),
76 matrix_free_data->get_constraint_vector(),
77 matrix_free_data->get_quadrature_vector(),
78 matrix_free_data->data);
79
80 // setup PDE operator and solver
81 pde_operator->setup(matrix_free, matrix_free_data);
82 pde_operator->setup_solver();
83
84 // finally, initialize dof vector
85 pde_operator->initialize_dof_vector(displacement);
86 }
87
88 std::shared_ptr<Poisson::Operator<dim, dim, Number> const>
89 get_pde_operator() const
90 {
91 return pde_operator;
92 }
93
94 std::shared_ptr<dealii::MatrixFree<dim, Number> const>
95 get_matrix_free() const
96 {
97 return matrix_free;
98 }
99
103 void
104 update(double const time,
105 bool const print_solver_info,
106 types::time_step time_step_number) override
107 {
108 // preconditioner update has no effect
109 (void)time_step_number;
110
111 dealii::Timer timer;
112 timer.restart();
113
114 VectorType rhs;
115 pde_operator->initialize_dof_vector(rhs);
116
117 // compute rhs and solve mesh deformation problem
118 pde_operator->rhs(rhs, time);
119
120 auto const n_iter = pde_operator->solve(displacement, rhs, time);
121 iterations.first += 1;
122 iterations.second += n_iter;
123
124 if(print_solver_info)
125 {
126 this->pcout << std::endl << "Solve moving mesh problem (Poisson):";
127 print_solver_info_linear(pcout, n_iter, timer.wall_time());
128 }
129
130 this->initialize_mapping_q_cache(this->mapping_undeformed,
131 displacement,
132 pde_operator->get_dof_handler());
133 }
134
138 void
139 print_iterations() const override
140 {
141 std::vector<std::string> names;
142 std::vector<double> iterations_avg;
143
144 names = {"Linear iterations"};
145 iterations_avg.resize(1);
146 iterations_avg[0] = (double)iterations.second / std::max(1., (double)iterations.first);
147
148 print_list_of_iterations(pcout, names, iterations_avg);
149 }
150
151private:
152 // matrix-free
153 std::shared_ptr<MatrixFreeData<dim, Number>> matrix_free_data;
154 std::shared_ptr<dealii::MatrixFree<dim, Number>> matrix_free;
155
156 // PDE operator
157 std::shared_ptr<Poisson::Operator<dim, dim, Number>> pde_operator;
158
159 // store solution of previous time step / iteration so that a good initial
160 // guess is available in the next step, easing convergence or reducing computational
161 // costs by allowing larger tolerances
162 VectorType displacement;
163
164 dealii::ConditionalOStream pcout;
165
166 std::pair<unsigned int /* calls */, unsigned long long /* iteration counts */> iterations;
167};
