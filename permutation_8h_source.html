55 int
56 cmp(const void * a, const void * b)
57 {
58  int ia = *(int *)a;
59  int ib = *(int *)b;
60  return array_proc[ia] == array_proc[ib] ?
61  // same process
62  array_list[ia] < array_list[ib] ? -1 : array_list[ia] > array_list[ib] :
63  // different processes
64  array_proc[ia] == __rank ?
65  -1 :
66  array_proc[ib] == __rank ?
67  +1 :
68  array_proc[ia] < array_proc[ib] ? -1 : array_proc[ia] > array_proc[ib];
69 }

119  int start, end, start_, end_;
120  // ... my range of the space filling curve
121  MAP.getLocalRange(start, end);
122  // ... my rows for FFT
123  FFT.getLocalRange(start_, end_);
124  bsize = FFT.bsize;
125 
126  // data structures for determining the communication partners...
127  int has_length = (end - start) * pow(points, dim);
128  unsigned long int * has = new unsigned long int[has_length];
129  int * has_procs = new int[has_length];
130  send_buffer = new double[has_length * dim];
131  send_index = new int[has_length];
132 
133  int want_length = (end_ - start_) * pow(n * points, dim - 1);
134  unsigned long int * want = new unsigned long int[want_length];
135  int * want_procs = new int[want_length];
136  recv_buffer = new double[want_length * dim];
137  recv_index = new int[want_length];
138 
139  // S.1: determine all dofs this process posses (+procs)
140  for(int ii = start, counter = 0; ii < end; ii++)
141  {
142  int i = MAP.indices(ii);
143  for(unsigned long int K = 0; K < (dim == 3 ? points : 1); K++)
144  for(unsigned long int J = 0; J < points; J++)
145  for(unsigned long int I = 0; I < points; I++, counter++)
146  {
147  // ... determine dof
148  unsigned long int temp =
149  ((dim == 3 ? ((MAP.lbf(i * dim + 2) * points + K) * points * n) : 0) +
150  MAP.lbf(i * dim + 1) * points + J) *
151  points * n +
152  (MAP.lbf(i * dim + 0) * points + I);
153  has[counter] = temp;
154  // ... which process does need this dof?
155  int proc = FFT.indices_proc_rows(temp / pow(n * points, dim - 1));
156  has_procs[counter] = proc;
157  }
158  }

160 
161  // S.2: which processes needs how many dofs
162  std::map<int, int> has_map;
163  has_map[rank] = 0; // add this rank
164  for(int counter = 0; counter < has_length; counter++)
165  {
166  int key = has_procs[counter];
167  if(has_map.find(key) == has_map.end())
168  has_map[key] = 1;
169  else
170  has_map[key]++;
171  }

174  // S.3: determine local indices for collecting data
175  for(int i = 0; i < has_length; i++)
176  send_index[i] = i;
177  array_list = has;
178  array_proc = has_procs;
179  __rank = rank;
180 
181  qsort(send_index, has_length, sizeof(int), cmp);
182 
183 
184  // S.4: determine offsets
185  for(auto & i : has_map)
186  if(i.first == rank)
187  {
188  send_offset.insert(send_offset.begin(), i.second);
189  send_procs.insert(send_procs.begin(), i.first);
190  }
191  else
192  {
193  send_offset.push_back(i.second);
194  send_procs.push_back(i.first);
195  }

197  send_offset.insert(send_offset.begin(), 0);
198 
199  std::partial_sum(send_offset.begin(), send_offset.end(), send_offset.begin());
200 
201 
202 
203  // R.1: determine all dofs this process wants (+procs)
204  // ... loop over all local rows
205  for(int ii = start_, pn = points * n, counter = 0; ii < end_; ii++)
206  // ... loop inside row
207  for(int jj = 0; jj < pow(n * points, dim - 1); jj++, counter++)
208  {
209  // ... determine dof
210  unsigned long int temp = ii * pow(n * points, dim - 1) + jj;
211  want[counter] = temp;
212  // ... determine owning process
213  int t = dim == 3 ? (temp % pn) / points + ((temp % (pn * pn)) / pn) / points * n +
214  (temp / pn / pn) / points * n * n :
215  (temp % pn) / points + (temp / pn) / points * n;
216  int proc = MAP.indices_proc(MAP.indices_inv(t));
217  want_procs[counter] = proc;
218  }
219 
220 
221  // R.2: which processes needs how many dofs
222  std::map<int, int> want_map;
223  want_map[rank] = 0; // add this rank
224  for(int counter = 0; counter < want_length; counter++)
225  {
226  int key = want_procs[counter];
227  if(want_map.find(key) == want_map.end())
228  want_map[key] = 1;
229  else
230  want_map[key]++;
231  }
232 
233  // R.3: determine local indices for distributing received data
234  for(int i = 0; i < want_length; i++)
235  recv_index[i] = i;
236  array_list = want;
237  array_proc = want_procs;
238 
239  qsort(recv_index, want_length, sizeof(int), cmp);
240 
241  // R.4: add padding
242  {
243  // consider padding needed by FFTW...
244  int N = points * n;
245  int Nx = (N / 2 + 1) * 2;
246  for(int i = 0; i < want_length; i++)
247  recv_index[i] += (recv_index[i] / N) * (Nx - N);
248  }
249 
250  // R.5: determine offsets
251  for(auto & i : want_map)
252  {
253  if(i.first == rank)
254  {
255  recv_offset.insert(recv_offset.begin(), i.second);
256  recv_procs.insert(recv_procs.begin(), i.first);
257  }
258  else
259  {
260  recv_offset.push_back(i.second);
261  recv_procs.push_back(i.first);
262  }
263  }
264 
265  recv_offset.insert(recv_offset.begin(), 0);
266 
267  std::partial_sum(recv_offset.begin(), recv_offset.end(), recv_offset.begin());
268 
269  // allocate memory for requests
270  recv_requests = new MPI_Request[recv_procs.size()];
271  send_requests = new MPI_Request[send_procs.size()];

401  void
402  ipermute(double * source, double * target)
403  {
404  // save references to vectors (for iwait)
405  this->source = source;
406  this->target = target;
407 
408  // request gathering of data ...
409  for(unsigned int i = 1; i < recv_procs.size(); i++)
410  MPI_Irecv(recv_buffer + recv_offset[i] * dim,
411  (recv_offset[i + 1] - recv_offset[i]) * dim,
412  MPI_DOUBLE,
413  recv_procs[i],
414  0,
415  comm,
416  recv_requests + i - 1);
417 
418  // scatter data ...
419  for(unsigned int i = 1; i < send_procs.size(); i++)
420  {
421  // ... copy data into buffer and ...
422  for(int j = send_offset[i]; j < send_offset[i + 1]; j++)
423  for(int d = 0; d < dim; d++)
424  send_buffer[dim * j + d] = source[send_index[j] * dim + d];
425 
426  // ... send right away
427  MPI_Isend(send_buffer + send_offset[i] * dim,
428  (send_offset[i + 1] - send_offset[i]) * dim,
429  MPI_DOUBLE,
430  send_procs[i],
431  0,
432  comm,
433  send_requests + i - 1);
434  }
435 
436  // perform copy operation on local process, if necessary
437  for(int s = send_offset[0], t = recv_offset[0]; s < send_offset[1]; s++, t++)
438  for(int d = 0; d < dim; d++)
439  target[recv_index[t] + d * bsize] = source[dim * send_index[s] + d];
440  }

446  void
447  iwait()
448  {
449  // wait for messages...
450  for(unsigned int c = 1; c < recv_procs.size(); c++)
451  {
452  int i = 0;
453  // ... a message received: extract index
454  MPI_Waitany(recv_procs.size() - 1, recv_requests, &i, MPI_STATUSES_IGNORE);
455  // ... copy data from buffers
456  for(int j = recv_offset[i + 1]; j < recv_offset[i + 2]; j++)
457  for(int d = 0; d < dim; d++)
458  target[recv_index[j] + d * bsize] = recv_buffer[j * dim + d];
459  }
460 
461  // wait that e.th has been send away
462  MPI_Waitall(send_procs.size() - 1, send_requests, MPI_STATUSES_IGNORE);
463  }
