55template<int dim, typename Number>
56class Grid
57{
58public:
59 typedef
60 typename std::vector<GridTools::PeriodicFacePair<typename Triangulation<dim>::cell_iterator>>
61 PeriodicFaces;
62
66 Grid(GridData const & data, MPI_Comm const & mpi_comm)
67 {
68 // triangulation
69 if(data.triangulation_type == TriangulationType::Serial)
70 {
71 AssertDimension(Utilities::MPI::n_mpi_processes(mpi_comm), 1);
72 triangulation = std::make_shared<Triangulation<dim>>();
73 }
74 else if(data.triangulation_type == TriangulationType::Distributed)
75 {
76 triangulation = std::make_shared<parallel::distributed::Triangulation<dim>>(
77 mpi_comm,
78 dealii::Triangulation<dim>::none,
79 parallel::distributed::Triangulation<dim>::construct_multigrid_hierarchy);
80 }
81 else if(data.triangulation_type == TriangulationType::FullyDistributed)
82 {
83 triangulation = std::make_shared<parallel::fullydistributed::Triangulation<dim>>(mpi_comm);
84 }
85 else
86 {
87 AssertThrow(false, ExcMessage("Invalid parameter triangulation_type."));
88 }
89
90 // mapping
91 mapping = std::make_shared<MappingQGeneric<dim>>(data.mapping_degree);
92 }
93
97 void
98 attach_grid_motion(std::shared_ptr<GridMotionInterface<dim, Number>> grid_motion_in)
99 {
100 grid_motion = grid_motion_in;
101 }
102
106 std::shared_ptr<Mapping<dim> const>
107 get_static_mapping() const
108 {
109 AssertThrow(mapping.get() != 0, ExcMessage("Grid::mapping is uninitialized."));
110
111 return mapping;
112 }
113
118 std::shared_ptr<Mapping<dim> const>
119 get_dynamic_mapping() const
120 {
121 if(grid_motion.get() != 0)
122 return grid_motion->get_mapping();
123 else
124 return mapping;
125 }
126
130 std::shared_ptr<Triangulation<dim>> triangulation;
131
135 PeriodicFaces periodic_faces;
136
140 std::shared_ptr<Mapping<dim>> mapping;

145 std::shared_ptr<GridMotionInterface<dim, Number>> grid_motion;
146};
