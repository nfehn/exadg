34template<int dim, typename Number>
35class GridMotionBase : public GridMotionInterface<dim, Number>
36{
37public:
38 typedef dealii::LinearAlgebra::distributed::Vector<Number> VectorType;
39
43 GridMotionBase(std::shared_ptr<dealii::Mapping<dim> const> mapping_undeformed,
44 unsigned int const mapping_degree_q_cache,
45 dealii::Triangulation<dim> const & triangulation)
46 : mapping_undeformed(mapping_undeformed)
47 {
48 // Make sure that dealii::MappingQCache is initialized correctly. An empty dof-vector is used
49 // and, hence, no displacements are added to the reference configuration described by
50 // mapping_undeformed.
51 dealii::DoFHandler<dim> dof_handler(triangulation);
52 VectorType displacement_vector;
53 moving_mapping = std::make_shared<MappingDoFVector<dim, Number>>(mapping_degree_q_cache);
54 moving_mapping->initialize_mapping_q_cache(mapping_undeformed,
55 displacement_vector,
56 dof_handler);
57 }
58
59 void
60 fill_grid_coordinates_vector(VectorType & grid_coordinates,
61 dealii::DoFHandler<dim> const & dof_handler) const final
62 {
63 moving_mapping->fill_grid_coordinates_vector(grid_coordinates, dof_handler);
64 }
65
66 std::shared_ptr<dealii::Mapping<dim> const>
67 get_mapping() const final
68 {
69 return moving_mapping;
70 }
71
72protected:
73 // mapping describing undeformed reference state
74 std::shared_ptr<dealii::Mapping<dim> const> mapping_undeformed;
75
76 // time-dependent mapping describing deformed state
77 std::shared_ptr<MappingDoFVector<dim, Number>> moving_mapping;
78};
