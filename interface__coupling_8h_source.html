43 template<int dim, int spacedim>
44 class InterfaceCommunicator
45 {
46 public:
47  /*
48  * Constructor.
49  */
50  InterfaceCommunicator(const std::vector<Point<spacedim>> & quadrature_points,
51  const parallel::TriangulationBase<dim, spacedim> & tria,
52  const Mapping<dim, spacedim> & mapping,
53  double const tolerance,
54  const std::vector<bool> & marked_vertices,
55  std::shared_ptr<GridTools::Cache<dim, dim>> const cache)
56  : comm(tria.get_communicator())
57  {
58  // create bounding boxed of local active cells
59  std::vector<BoundingBox<spacedim>> local_boxes;
60  for(auto const cell : tria.active_cell_iterators())
61  if(cell->is_locally_owned())
62  local_boxes.push_back(mapping.get_bounding_box(cell));
63 
64  // it might happen that there are no cells on a processor
65  if(local_boxes.size() == 0)
66  local_boxes.resize(1);
67 
68  // create r-tree of bounding boxes
69  auto const local_tree = pack_rtree(local_boxes);
70 
71  // compress r-tree to a minimal set of bounding boxes
72  auto const local_reduced_box = extract_rtree_level(local_tree, 0);
73 
74  // gather bounding boxes of other processes
75  auto const global_bounding_boxes = Utilities::MPI::all_gather(comm, local_reduced_box);
76 
77  // determine ranks which might poses quadrature point
78  auto points_per_process =
79  std::vector<std::vector<Point<spacedim>>>(global_bounding_boxes.size());
80 
81  auto points_per_process_offset =
82  std::vector<std::vector<unsigned int>>(global_bounding_boxes.size());
83 
84  for(unsigned int i = 0; i < quadrature_points.size(); ++i)
85  {
86  auto const & point = quadrature_points[i];
87  for(unsigned rank = 0; rank < global_bounding_boxes.size(); ++rank)
88  for(auto const & box : global_bounding_boxes[rank])
89  if(box.point_inside(point, tolerance))
90  {
91  points_per_process[rank].emplace_back(point);
92  points_per_process_offset[rank].emplace_back(i);
93  break;
94  }
95  }
96 
97  // only communicate with processes that might have a quadrature point
98  std::vector<unsigned int> targets;
99 
100  for(unsigned int i = 0; i < points_per_process.size(); ++i)
101  if(points_per_process[i].size() > 0 && i != Utilities::MPI::this_mpi_process(comm))
102  targets.emplace_back(i);
103 
104 
105  std::map<unsigned int, std::vector<Point<spacedim>>> relevant_points_per_process;
106  std::map<unsigned int, std::vector<unsigned int>> relevant_points_per_process_offset;
107  std::map<unsigned int, std::vector<unsigned int>> relevant_points_per_process_count;
108 
109 
110  // for local quadrature points no communication is needed...
111  {
112  unsigned int const my_rank = Utilities::MPI::this_mpi_process(comm);
113 
114  auto const & potentially_local_points = points_per_process[my_rank];
115 
116 
117  {
118  std::vector<Point<spacedim>> points;
119  std::vector<unsigned int> points_offset;
120  std::vector<unsigned int> count;
121 
122  auto const & potentially_relevant_points = points_per_process[my_rank];
123  auto const & potentially_relevant_points_offset = points_per_process_offset[my_rank];
124 
125  typename Triangulation<dim, dim>::active_cell_iterator cell_hint =
126  typename Triangulation<dim, dim>::active_cell_iterator();
127  for(unsigned int j = 0; j < potentially_local_points.size(); ++j)
128  {
129  unsigned int const counter =
130  n_locally_owned_active_cells_around_point(tria,
131  mapping,
132  potentially_relevant_points[j],
133  tolerance,
134  cell_hint,
135  marked_vertices,
136  *cache);
137 
138  if(counter > 0)
139  {
140  points.push_back(potentially_relevant_points[j]);
141  points_offset.push_back(potentially_relevant_points_offset[j]);
142  count.push_back(counter);
143  }
144  }
145 
146 
147  if(points.size() > 0)
148  {
149  relevant_remote_points_per_process[my_rank] = points;
150  relevant_remote_points_count_per_process[my_rank] = count;
151 
152  relevant_points_per_process[my_rank] = points;
153  relevant_points_per_process_offset[my_rank] = points_offset;
154  relevant_points_per_process_count[my_rank] = count;
155  map_recv[my_rank] = points;
156  }
157  }
158  }
159 
160  // send to remote ranks the requested quadrature points and eliminate not needed ones
161  // (note: currently, we cannot communicate points -> switch to doubles here)
162  Utilities::MPI::ConsensusAlgorithms::AnonymousProcess<double, unsigned int> process(
163  [&]() { return targets; },
164  [&](unsigned int const other_rank, std::vector<double> & send_buffer) {
165  // send requested points
166  for(auto point : points_per_process[other_rank])
167  for(unsigned int i = 0; i < spacedim; ++i)
168  send_buffer.emplace_back(point[i]);
169  },
170  [&](unsigned int const & other_rank,
171  const std::vector<double> & recv_buffer,
172  std::vector<unsigned int> & request_buffer) {
173  // received points, determine if point is actually possessed, and
174  // send the result back
175 
176  std::vector<Point<spacedim>> relevant_remote_points;
177  std::vector<unsigned int> relevant_remote_points_count;
178 
179  request_buffer.clear();
180  request_buffer.resize(recv_buffer.size() / spacedim);
181 
182  typename Triangulation<dim, dim>::active_cell_iterator cell_hint =
183  typename Triangulation<dim, dim>::active_cell_iterator();
184  for(unsigned int i = 0, j = 0; i < recv_buffer.size(); i += spacedim, ++j)
185  {
186  Point<spacedim> point;
187  for(unsigned int j = 0; j < spacedim; ++j)
188  point[j] = recv_buffer[i + j];
189 
190  unsigned int const counter = n_locally_owned_active_cells_around_point(
191  tria, mapping, point, tolerance, cell_hint, marked_vertices, *cache);
192 
193  request_buffer[j] = counter;
194 
195  if(counter > 0)
196  {
197  relevant_remote_points.push_back(point);
198  relevant_remote_points_count.push_back(counter);
199  }
200  }
201 
202  if(relevant_remote_points.size() > 0)
203  {
204  relevant_remote_points_per_process[other_rank] = relevant_remote_points;
205  relevant_remote_points_count_per_process[other_rank] = relevant_remote_points_count;
206  }
207  },
208  [&](unsigned int const other_rank, std::vector<unsigned int> & recv_buffer) {
209  // prepare buffer
210  recv_buffer.resize(points_per_process[other_rank].size());
211  },
212  [&](unsigned int const other_rank, const std::vector<unsigned int> & recv_buffer) {
213  // store recv_buffer -> make the algorithm deterministic
214 
215  auto const & potentially_relevant_points = points_per_process[other_rank];
216  auto const & potentially_relevant_points_offset = points_per_process_offset[other_rank];
217 
218  std::vector<Point<spacedim>> points;
219  std::vector<unsigned int> points_offset;
220  std::vector<unsigned int> count;
221 
222  AssertDimension(potentially_relevant_points.size(), recv_buffer.size());
223  AssertDimension(potentially_relevant_points_offset.size(), recv_buffer.size());
224 
225  for(unsigned int i = 0; i < recv_buffer.size(); ++i)
226  if(recv_buffer[i] > 0)
227  {
228  points.push_back(potentially_relevant_points[i]);
229  points_offset.push_back(potentially_relevant_points_offset[i]);
230  count.push_back(recv_buffer[i]);
231  }
232 
233  if(points.size() > 0)
234  {
235  relevant_points_per_process[other_rank] = points;
236  relevant_points_per_process_offset[other_rank] = points_offset;
237  relevant_points_per_process_count[other_rank] = count;
238  map_recv[other_rank] = points;
239  }
240  });
241 
242  Utilities::MPI::ConsensusAlgorithms::Selector<double, unsigned int>(process, comm).run();
243 
244  quadrature_points_count.resize(quadrature_points.size(), 0);
245 
246  for(auto const & i : relevant_points_per_process)
247  {
248  unsigned int const rank = i.first;
249 
250  std::vector<std::pair<unsigned int, unsigned int>> indices;
251 
252  auto const & relevant_points = relevant_points_per_process[rank];
253  auto const & relevant_points_offset = relevant_points_per_process_offset[rank];
254  auto const & relevant_points_count = relevant_points_per_process_count[rank];
255 
256  for(unsigned int j = 0; j < relevant_points.size(); ++j)
257  {
258  for(unsigned int k = 0; k < relevant_points_count[j]; ++k)
259  {
260  AssertIndexRange(relevant_points_offset[j], quadrature_points_count.size());
261  auto & qp_counter = quadrature_points_count[relevant_points_offset[j]];
262  indices.emplace_back(relevant_points_offset[j], qp_counter);
263 
264  ++qp_counter;
265  }
266  }
267 
268  this->indices_per_process[rank] = indices;
269  }
270  }

272  template<typename T>
273  void
274  process(const std::map<unsigned int, std::vector<std::vector<T>>> & input,
275  std::vector<std::vector<T>> & output,
276  unsigned int const tag) const
277  {
278  // process remote quadrature points and send them away
279  std::map<unsigned int, std::vector<char>> temp_map;
280 
281  std::vector<MPI_Request> requests;
282  requests.reserve(input.size());
283 
284  unsigned int const my_rank = Utilities::MPI::this_mpi_process(comm);
285 
286  for(auto const & vec : input)
287  {
288  if(vec.first == my_rank)
289  continue;
290 
291  temp_map[vec.first] = Utilities::pack(vec.second);
292 
293  auto & buffer = temp_map[vec.first];
294 
295  requests.resize(requests.size() + 1);
296 
297  MPI_Isend(buffer.data(), buffer.size(), MPI_CHAR, vec.first, tag, comm, &requests.back());
298  }
299 
300  // receive result
301 
302  std::map<unsigned int, std::vector<std::vector<T>>> temp_recv_map;
303 
304  // process locally-owned values
305  if(input.find(my_rank) != input.end())
306  temp_recv_map[my_rank] = input.at(my_rank);
307 
308  for(auto const & vec : map_recv)
309  {
310  if(vec.first == my_rank)
311  continue;
312 
313  MPI_Status status;
314  MPI_Probe(MPI_ANY_SOURCE, tag, comm, &status);
315 
316  int message_length;
317  MPI_Get_count(&status, MPI_CHAR, &message_length);
318 
319  std::vector<char> buffer(message_length);
320 
321  MPI_Recv(
322  buffer.data(), buffer.size(), MPI_CHAR, status.MPI_SOURCE, tag, comm, MPI_STATUS_IGNORE);
323 
324  temp_recv_map[status.MPI_SOURCE] = Utilities::unpack<std::vector<std::vector<T>>>(buffer);
325  }
326 
327  MPI_Waitall(requests.size(), requests.data(), MPI_STATUSES_IGNORE);
328 
329  for(auto const & i : temp_recv_map)
330  {
331  unsigned int const rank = i.first;
332 
333  Assert(indices_per_process.find(rank) != indices_per_process.end(),
334  ExcMessage("Map does not contain this rank."));
335 
336  auto it = indices_per_process.at(rank).begin();
337 
338  for(auto const & i : temp_recv_map[rank])
339  for(auto const & j : i)
340  {
341  output[it->first][it->second] = j;
342  ++it;
343  }
344  }
345  }

348  void
349  init_solution_values(std::vector<std::vector<T>> & output) const
350  {
351  output.resize(quadrature_points_count.size());

354  output[i] = std::vector<T>(quadrature_points_count[i], T());

357  template<typename T>
358  void
359  init_remote_solution_values(std::map<unsigned int, std::vector<std::vector<T>>> & input) const
360  {
361  for(auto const & i : this->relevant_remote_points_count_per_process)
362  {
363  unsigned int const rank = i.first;
364 
365  std::vector<std::vector<T>> temp(i.second.size());
366 
367  for(unsigned int j = 0; j < i.second.size(); ++j)
368  temp[j] = std::vector<T>(i.second[j], T());
369 
370  input[rank] = temp;
371  }
372  }

383 private:
384  const MPI_Comm & comm;
385 
386  // receiver side
387  std::vector<unsigned int> quadrature_points_count;
388  std::map<unsigned int, std::vector<std::pair<unsigned int, unsigned int>>> indices_per_process;
389  std::map<unsigned int, std::vector<Point<spacedim>>> map_recv;
390 
391  // sender side
392  std::map<unsigned int, std::vector<Point<spacedim>>> relevant_remote_points_per_process;
393  std::map<unsigned int, std::vector<unsigned int>> relevant_remote_points_count_per_process;
394 };

397 template<int dim, int n_components, typename Number>
398 class InterfaceCoupling
399 {
400 private:
401  static unsigned int const rank =
402  (n_components == 1) ? 0 : ((n_components == dim) ? 1 : numbers::invalid_unsigned_int);
403 
404  typedef InterfaceCoupling<dim, n_components, Number> This;
405 
406  typedef LinearAlgebra::distributed::Vector<Number> VectorType;
407  typedef LinearAlgebra::distributed::Vector<double> VectorTypeDouble;
408  typedef FaceIntegrator<dim, n_components, Number> Integrator;
409  typedef std::pair<unsigned int, unsigned int> Range;
410 
411  typedef std::map<types::boundary_id, std::shared_ptr<FunctionCached<rank, dim, double>>>
412  MapBoundaryCondition;
413 
414  typedef unsigned int quad_index;
415  typedef unsigned int mpi_rank;
416 
417  typedef std::tuple<unsigned int /*face*/, unsigned int /*q*/, unsigned int /*v*/> Id;
418 
419  typedef std::map<Id, types::global_dof_index> MapIndex;
420 
421  typedef std::pair<std::vector<types::global_dof_index>, std::vector<double>> Cache;
422 
423  typedef std::vector<Point<dim>> ArrayQuadraturePoints;
424  typedef std::vector<std::vector<Cache>> ArrayVectorCache;
425  typedef std::vector<std::vector<Tensor<rank, dim, double>>> ArrayVectorTensor;
426 
427 public:
428  InterfaceCoupling() : dof_index_dst(0), dof_handler_src(nullptr), mapping_src(nullptr)
429  {
430  }
431 
432  void
433  setup(std::shared_ptr<MatrixFree<dim, Number>> matrix_free_dst_in,
434  unsigned int const dof_index_dst_in,
435  std::vector<quad_index> const & quad_indices_dst_in,
436  MapBoundaryCondition const & map_bc_in,
437  std::shared_ptr<parallel::TriangulationBase<dim>> const triangulation_src_in,
438  DoFHandler<dim> const & dof_handler_src_in,
439  Mapping<dim> const & mapping_src_in,
440  VectorType const & dof_vector_src_in,
441  double const tolerance)
442  {
443  matrix_free_dst = matrix_free_dst_in;
444  dof_index_dst = dof_index_dst_in;
445  quad_rules_dst = quad_indices_dst_in;
446  map_bc = map_bc_in;
447  triangulation_src = triangulation_src_in;
448  dof_handler_src = &dof_handler_src_in;
449  mapping_src = &mapping_src_in;
450 
451  std::shared_ptr<GridTools::Cache<dim, dim>> cache_src;
452  cache_src.reset(new GridTools::Cache<dim, dim>(*triangulation_src, *mapping_src));
453 
454  // mark vertices at interface in order to make search of active cells around point more
455  // efficient
456  std::vector<bool> marked_vertices(triangulation_src->n_vertices(), false);
457  for(auto cell : triangulation_src->active_cell_iterators())
458  {
459  if(!cell->is_artificial() && cell->at_boundary())
460  {
461  for(unsigned int f = 0; f < GeometryInfo<dim>::faces_per_cell; ++f)
462  {
463  if(cell->face(f)->at_boundary())
464  {
465  if(map_bc.find(cell->face(f)->boundary_id()) != map_bc.end())
466  {
467  for(unsigned int v = 0; v < GeometryInfo<dim - 1>::vertices_per_cell; ++v)
468  {
469  marked_vertices[cell->face(f)->vertex_index(v)] = true;
470  }
471  }
472  }
473  }
474  }
475  }
476 
477  // implementation needs Number = double
478  VectorTypeDouble dof_vector_src_double_copy;
479  VectorTypeDouble const * dof_vector_src_double_ptr;
480  if(std::is_same<double, Number>::value)
481  {
482  dof_vector_src_double_ptr = reinterpret_cast<VectorTypeDouble const *>(&dof_vector_src_in);
483  }
484  else
485  {
486  dof_vector_src_double_copy = dof_vector_src_in;
487  dof_vector_src_double_ptr = &dof_vector_src_double_copy;
488  }
489 
490  for(auto quadrature : quad_rules_dst)
491  {
492  // initialize maps
493  map_index_dst.emplace(quadrature, MapIndex());
494  map_q_points_dst.emplace(quadrature, ArrayQuadraturePoints());
495  map_solution_dst.emplace(quadrature, ArrayVectorTensor());
496 
497  MapIndex & map_index = map_index_dst.find(quadrature)->second;
498  ArrayQuadraturePoints & array_q_points_dst = map_q_points_dst.find(quadrature)->second;
499  ArrayVectorTensor & array_solution_dst = map_solution_dst.find(quadrature)->second;
500 
501 
502  /*
503  * 1. Setup: create map "ID = {face, q, v} <-> vector_index" and fill array of quadrature
504  * points
505  */
506  for(unsigned int face = matrix_free_dst->n_inner_face_batches();
507  face <
508  matrix_free_dst->n_inner_face_batches() + matrix_free_dst->n_boundary_face_batches();
509  ++face)
510  {
511  // only consider relevant boundary IDs
512  if(map_bc.find(matrix_free_dst->get_boundary_id(face)) != map_bc.end())
513  {
514  Integrator integrator(*matrix_free_dst, true, dof_index_dst, quadrature);
515  integrator.reinit(face);
516 
517  for(unsigned int q = 0; q < integrator.n_q_points; ++q)
518  {
519  Point<dim, VectorizedArray<Number>> q_points = integrator.quadrature_point(q);
520 
521  for(unsigned int v = 0; v < VectorizedArray<Number>::size(); ++v)
522  {
523  Point<dim> q_point;
524  for(unsigned int d = 0; d < dim; ++d)
525  q_point[d] = q_points[d][v];
526 
527  Id id = std::make_tuple(face, q, v);
528  types::global_dof_index index = array_q_points_dst.size();
529  map_index.emplace(id, index);
530  array_q_points_dst.push_back(q_point);
531  }
532  }
533  }
534  }
535 
536  map_q_points_src.emplace(quadrature, std::map<mpi_rank, ArrayQuadraturePoints>());
537  map_cache_src.emplace(quadrature, std::map<mpi_rank, ArrayVectorCache>());
538  map_solution_src.emplace(quadrature, std::map<mpi_rank, ArrayVectorTensor>());
539 
540  std::map<mpi_rank, ArrayQuadraturePoints> & mpi_map_q_points_src =
541  map_q_points_src.find(quadrature)->second;
542  std::map<mpi_rank, ArrayVectorCache> & mpi_map_cache_src =
543  map_cache_src.find(quadrature)->second;
544  std::map<mpi_rank, ArrayVectorTensor> & mpi_map_solution_src =
545  map_solution_src.find(quadrature)->second;
546 
547 
548  /*
549  * 2. Communication: receive and cache quadrature points of other ranks,
550  * redundantly store own q-points (those that are needed)
551  */
552  map_communicator.emplace(quadrature,
553  InterfaceCommunicator<dim, dim>(array_q_points_dst,
554  *triangulation_src,
555  *mapping_src,
556  tolerance,
557  marked_vertices,
558  cache_src));
559 
560  InterfaceCommunicator<dim, dim> & communicator = map_communicator.find(quadrature)->second;
561 
562  mpi_map_q_points_src = communicator.get_remote_quadrature_points();
563 
564  communicator.init_solution_values(array_solution_dst);
565 
566  communicator.init_remote_solution_values(mpi_map_solution_src);
567 
568 
569  /*
570  * 3. Compute dof indices and shape values for all quadrature points
571  */
572  typename Triangulation<dim, dim>::active_cell_iterator cell_hint =
573  triangulation_src->begin_active();
574 
575  for(auto it : mpi_map_q_points_src)
576  {
577  mpi_rank const proc = it.first;
578  ArrayQuadraturePoints & array_q_points_src = it.second;
579 
580  mpi_map_cache_src.emplace(proc, ArrayVectorCache());
581  ArrayVectorCache & array_cache_src = mpi_map_cache_src.find(proc)->second;
582  array_cache_src.resize(array_q_points_src.size());
583 
584  for(types::global_dof_index q = 0; q < array_q_points_src.size(); ++q)
585  {
586  auto adjacent_cells =
587  find_all_active_cells_around_point(*mapping_src,
588  dof_handler_src->get_triangulation(),
589  array_q_points_src[q],
590  tolerance,
591  cell_hint,
592  marked_vertices,
593  *cache_src);
594 
595  array_cache_src[q] = get_dof_indices_and_shape_values(adjacent_cells,
596  *dof_handler_src,
597  *mapping_src,
598  *dof_vector_src_double_ptr);
599 
600  AssertThrow(array_cache_src[q].size() > 0,
601  ExcMessage("No adjacent points have been found."));
602  }
603  }
604 
605 
606  /*
607  * 4. Communication: transfer results back to dst-side
608  */
609  communicator.process(mpi_map_solution_src, array_solution_dst, 11 + quadrature);
610  }
611 
612  // finally, give boundary condition access to the data
613  for(auto boundary : map_bc)
614  {
615  boundary.second->set_data_pointer(map_index_dst, map_solution_dst);
616  }
617  }

619  void
620  update_data(VectorType const & dof_vector_src)
621  {
622  VectorTypeDouble dof_vector_src_double_copy;
623  VectorTypeDouble const * dof_vector_src_double_ptr;
