31 /*
32  * This function wraps the copy of a PETSc object (sparse matrix,
33  * preconditioner) with a dealii::LinearAlgebra::distributed::Vector.
34  */
35 #ifdef DEAL_II_WITH_PETSC
36 template<typename VectorType>
37 void
38 apply_petsc_operation(VectorType & dst,
39  VectorType const & src,
40  std::function<void(PETScWrappers::VectorBase &,
41  PETScWrappers::VectorBase const &)> petsc_operation)
42 {
43  // copy to petsc internal vector type because there is currently no such
44  // function in deal.II (and the transition via ReadWriteVector is too
45  // slow/poorly tested)
46  Vec vector_dst, vector_src;
47  VecCreateMPI(dst.get_mpi_communicator(),
48  dst.get_partitioner()->local_size(),
49  PETSC_DETERMINE,
50  &vector_dst);
51  VecCreateMPI(src.get_mpi_communicator(),
52  src.get_partitioner()->local_size(),
53  PETSC_DETERMINE,
54  &vector_src);
55 
56  {
57  PetscInt begin, end;
58  PetscErrorCode ierr = VecGetOwnershipRange(vector_src, &begin, &end);
59  AssertThrow(ierr == 0, ExcPETScError(ierr));
60 
61  PetscScalar * ptr;
62  ierr = VecGetArray(vector_src, &ptr);
63  AssertThrow(ierr == 0, ExcPETScError(ierr));
64 
65  const PetscInt local_size = src.get_partitioner()->local_size();
66  AssertDimension(local_size, static_cast<unsigned int>(end - begin));
67  for(PetscInt i = 0; i < local_size; ++i)
68  {
69  ptr[i] = src.local_element(i);
70  }
71 
72  ierr = VecRestoreArray(vector_src, &ptr);
73  AssertThrow(ierr == 0, ExcPETScError(ierr));
74  }
75 
76  // wrap `Vec` into VectorBase (without copying data)
77  PETScWrappers::VectorBase petsc_dst(vector_dst);
78  PETScWrappers::VectorBase petsc_src(vector_src);
79 
80  petsc_operation(petsc_dst, petsc_src);
81 
82  {
83  PetscInt begin, end;
84  PetscErrorCode ierr = VecGetOwnershipRange(vector_dst, &begin, &end);
85  AssertThrow(ierr == 0, ExcPETScError(ierr));
86 
87  PetscScalar * ptr;
88  ierr = VecGetArray(vector_dst, &ptr);
89  AssertThrow(ierr == 0, ExcPETScError(ierr));
90 
91  const PetscInt local_size = dst.get_partitioner()->local_size();
92  AssertDimension(local_size, static_cast<unsigned int>(end - begin));
93 
94  for(PetscInt i = 0; i < local_size; ++i)
95  {
96  dst.local_element(i) = ptr[i];
97  }
98 
99  ierr = VecRestoreArray(vector_dst, &ptr);
100  AssertThrow(ierr == 0, ExcPETScError(ierr));
101  }
102 }
103 #endif
