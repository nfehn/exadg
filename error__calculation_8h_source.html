38 template<int dim>
39 struct ErrorCalculationData

41  ErrorCalculationData()

43  calculate_relative_errors(true),
44  calculate_H1_seminorm_error(false),
45  error_calc_start_time(std::numeric_limits<double>::max()),

53 
54  void
55  print(ConditionalOStream & pcout, bool unsteady)
56  {
57  print_parameter(pcout, "Calculate error", analytical_solution_available);
58  if(analytical_solution_available == true && unsteady == true)
59  {
60  print_parameter(pcout, "Calculate relative errors", calculate_relative_errors);
61  print_parameter(pcout, "Calculate H1-seminorm error", calculate_H1_seminorm_error);
62  print_parameter(pcout, "Error calculation start time", error_calc_start_time);
63  print_parameter(pcout, "Error calculation interval time", error_calc_interval_time);
64  print_parameter(pcout, "Calculate error every time steps", calculate_every_time_steps);
65  print_parameter(pcout, "Write errors to file", write_errors_to_file);
66  print_parameter(pcout, "Output folder", folder);
67  print_parameter(pcout, "Name", name);
68  }
69  }
70 
71  // to calculate the error an analytical solution to the problem has to be available
72  bool analytical_solution_available;
73 
74  std::shared_ptr<Function<dim>> analytical_solution;
75 
76  // relative or absolute errors?
77  // If calculate_relative_errors == false, this implies that absolute errors are calculated
78  bool calculate_relative_errors;
79 
80  // by default, only the L2-error is computed. Other norms have to be explicitly specified by the
81  // user.
82  bool calculate_H1_seminorm_error;
83 
84  // before then no error calculation will be performed
85  double error_calc_start_time;
86 
87  // specifies the time interval in which error calculation is performed
88  double error_calc_interval_time;
89 
90  // calculate error every time steps
91  unsigned int calculate_every_time_steps;
92 
93  // write errors to file?
94  bool write_errors_to_file;
95 
96  // output folder and name (used as filename and as identifier for screen output)
97  std::string folder;
98  std::string name;
99 };
