194  void
195  interpolate(double const *& src)
196  {
197  // allocate dst- and src-vector
198  AlignedVector<double> temp1(MAX(dofs_source, dofs_target));
199  AlignedVector<double> temp2(MAX(dofs_source, dofs_target));
200 
201  // get start point of arrays
202  double const * src_ = src;
203  double * dst_ = dst;
204  // loop over all cells
205  for(int c = 0; c < cells; c++)
206  {
207  // loop over all velocity directions
208  for(int d = 0; d < DIM; d++)
209  {
210  // perform interpolation
211  if(DIM == 2)
212  {
213  // ... 2D
214  dealii::internal::
215  EvaluatorTensorProduct<dealii::internal::evaluate_general, 2, 0, 0, double>
216  eval_val(shape_values, shape_values, shape_values, points_source, points_target);
217  eval_val.template values<0, true, false>(src_, temp2.begin());
218  eval_val.template values<1, true, false>(temp2.begin(), temp1.begin());
219  }
220  else
221  {
222  // ... 3D
223  dealii::internal::
224  EvaluatorTensorProduct<dealii::internal::evaluate_general, 3, 0, 0, double>
225  eval_val(shape_values, shape_values, shape_values, points_source, points_target);
226  eval_val.template values<0, true, false>(src_, temp1.begin());
227  eval_val.template values<1, true, false>(temp1.begin(), temp2.begin());
228  eval_val.template values<2, true, false>(temp2.begin(), temp1.begin());
229  }
230  // write interpolated data permuted into destination vector
231  for(int i = 0; i < dofs_target; i++)
232  dst_[i * DIM + d] = temp1[i];
233  // go to next dimension
234  src_ += dofs_source;
235  }
236  // go to next cell
237  dst_ += dofs_target * DIM;
238  }
239  }

337  template<typename VEC>
338  void
339  io(int type, char const * filename, VEC & src)
340  {
341  // dofs to read/write per field
342  unsigned long int dofs = cells * dofs_source;
343  // local displacement in file (in bytes)
344  MPI_Offset disp =
345  8 * sizeof(int) + static_cast<unsigned long int>(start) * dofs_source * sizeof(double) * DIM;
346 
347  // create view
348  MPI_Datatype stype;
349  MPI_Type_contiguous(dofs, MPI_DOUBLE, &stype);
350  MPI_Type_commit(&stype);
351 
352  // ooen file ...
353  MPI_File fh;
354  if(type == 0)
355  MPI_File_open(comm, filename, MPI_MODE_RDONLY, MPI_INFO_NULL, &fh);
356  else
357  MPI_File_open(comm, filename, MPI_MODE_WRONLY, MPI_INFO_NULL, &fh);
358 
359  // ... set view
360  MPI_File_set_view(fh, disp, MPI_DOUBLE, MPI_DOUBLE, "native", MPI_INFO_NULL);
361 
362  if(type == 0)
363  // ... read file
364  MPI_File_read_all(fh, (void *)src, dofs * DIM, MPI_DOUBLE, MPI_STATUSES_IGNORE);
365  else
366  // ... write file
367  MPI_File_write_all(fh, src, dofs * DIM, MPI_DOUBLE, MPI_STATUSES_IGNORE);
368 
369  // ... close file
370  MPI_File_close(&fh);
371 
372  // clean up
373  MPI_Type_free(&stype);
374  }
