29 /*
30  * Note:
31  * The default argument "double" could be removed but this implies that all BoundaryDescriptors
32  * that use FunctionCached require another template parameter "Number", which requires
33  * changes of major parts of the code.
34  */
35 template<int rank, int dim, typename Number = double>
36 class FunctionCached
37 {
38 private:
39  typedef std::tuple<unsigned int /*face*/, unsigned int /*q*/, unsigned int /*v*/> Id;
40  typedef std::map<Id, types::global_dof_index> MapVectorIndex;
41  typedef std::vector<std::vector<Tensor<rank, dim, Number>>> ArrayVectorTensor;
42 
43 public:
44  FunctionCached() : global_map_vector_index(nullptr), map_solution(nullptr)
45  {
46  }
47 
48  Tensor<rank, dim, Number>
49  tensor_value(unsigned int const face,
50  unsigned int const q,
51  unsigned int const v,
52  unsigned int const quad_index) const
53  {
54  Assert(global_map_vector_index != nullptr,
55  ExcMessage("Pointer global_map_vector_index is not initialized."));
56  Assert(global_map_vector_index->find(quad_index) != global_map_vector_index->end(),
57  ExcMessage("Specified quad_index does not exist in global_map_vector_index."));
58 
59  Assert(map_solution != nullptr, ExcMessage("Pointer map_solution is not initialized."));
60  Assert(map_solution->find(quad_index) != map_solution->end(),
61  ExcMessage("Specified quad_index does not exist in map_solution."));
62 
63  MapVectorIndex const & map_vector_index = global_map_vector_index->find(quad_index)->second;
64  ArrayVectorTensor const & array_solution = map_solution->find(quad_index)->second;
65 
66  Id id = std::make_tuple(face, q, v);
67  types::global_dof_index index = map_vector_index.find(id)->second;
68 
69  Assert(index < array_solution.size(), ExcMessage("Index exceeds dimensions of vector."));
70 
71  std::vector<Tensor<rank, dim, double>> const & vector_solution = array_solution[index];
72 
73  // average
74  unsigned int counter = 0;
75  Tensor<rank, dim, double> solution;
76  for(unsigned int i = 0; i < vector_solution.size(); ++i)
77  {
78  solution += vector_solution[i];
79  ++counter;
80  }
81 
82  Assert(counter > 0, ExcMessage("Vector must contain at least one value."));
83 
84  solution *= 1.0 / (double)counter;
85 
86  return solution;
87  }
88 
89  void
90  set_data_pointer(std::map<unsigned int, MapVectorIndex> const & global_map_vector_index_in,
91  std::map<unsigned int, ArrayVectorTensor> const & map_solution_in)
92  {
93  global_map_vector_index = &global_map_vector_index_in;
94  map_solution = &map_solution_in;
95  }
96 
97 private:
98  std::map<unsigned int, MapVectorIndex> const * global_map_vector_index;
99  std::map<unsigned int, ArrayVectorTensor> const * map_solution;
100 };
