119  readHeader(char *& filename)
120  {
121  FILE * fp;
122  int crit[1 + 4];
123  crit[0] = 0;
124 
125  if(this->rank == 0)
126  {
127  // read header only by rank 0 ...
128  if((fp = fopen(filename, "r")) != NULL)
129  {
130  // ... read header in one go
131  fread(crit + 1, sizeof(int), 4, fp);
132 
133  // read time
134  fread(&this->time, sizeof(double), 1, fp);
135 
136  // ... close file
137  fclose(fp);
138  }
139  else
140  {
141  // ... reading the file failed
142  crit[0] = 0;
143  }
144  }
145 
146  // broadcast header to all processes
147  MPI_Bcast(&crit, 5, MPI_INT, 0, comm);
148  MPI_Bcast(&this->time, 1, MPI_DOUBLE, 0, comm);
149 
150  if(this->initialized)
151  {
152  // ... extract header by each process and set properties
153  crit[0] += this->type != crit[1];
154  crit[0] += this->dim != crit[2];
155  crit[0] += this->cells != crit[3];
156  crit[0] += this->degree != crit[4];
157  }
158  else
159  {
160  // ... extract header by each process and set properties
161  this->type = crit[1];
162  this->dim = crit[2];
163  this->cells = crit[3];
164  this->degree = crit[4];
165  this->points_src = this->degree + 1;
166  if(this->points_dst == 0)
167  this->points_dst = this->points_src; // equal if nothing specified
168  }
169 
170  // success or failure?
171  return crit[0];
172  }

203  int temp1 = this->cells * this->points_dst;
204  int temp2 = 0;
205  fwrite(&temp1, sizeof(int), temp1, fp);
206  fwrite(&temp2, sizeof(int), temp2, fp);
207  }
208 
209  // write time

211 
212  // ... close file
213  fclose(fp);
214  }
215  }
