144  /*
145  * Smagorinsky model (1963):
146  *
147  * nu_SGS = (C * filter_width)^{2} * sqrt(2 * S:S)
148  *
149  * where S is the symmetric part of the velocity gradient
150  *
151  * S = 1/2 * (grad(u) + grad(u)^T) and S:S = S_ij * S_ij
152  *
153  * and the model constant is
154  *
155  * C = 0.165 (Nicoud et al. (2011))
156  * C = 0.18 (Toda et al. (2010))
157  */
158  void
159  smagorinsky_model(scalar const & filter_width,
160  tensor const & velocity_gradient,
161  double const & C,
162  scalar & viscosity) const;
163 
164  /*
165  * Vreman model (2004): Note that we only consider the isotropic variant of the Vreman model:
166  *
167  * nu_SGS = (C * filter_width)^{2} * D
168  *
169  * where the differential operator D is defined as
170  *
171  * D = sqrt(B_gamma / ||grad(u)||^{2})
172  *
173  * with
174  *
175  * ||grad(u)||^{2} = grad(u) : grad(u) and grad(u) = d(u_i)/d(x_j) ,
176  *
177  * gamma = grad(u) * grad(u)^T ,
178  *
179  * and
180  *
181  * B_gamma = gamma_11 * gamma_22 - gamma_12^{2}
182  * +gamma_11 * gamma_33 - gamma_13^{2}
183  * +gamma_22 * gamma_33 - gamma_23^{2}
184  *
185  * Note that if ||grad(u)||^{2} = 0, nu_SGS is consistently defined as zero.
186  *
187  */
188  void
189  vreman_model(scalar const & filter_width,
190  tensor const & velocity_gradient,
191  double const & C,
192  scalar & viscosity) const;
193 
194  /*
195  * WALE (wall-adapting local eddy-viscosity) model (Nicoud & Ducros 1999):
196  *
197  * nu_SGS = (C * filter_width)^{2} * D ,
198  *
199  * where the differential operator D is defined as
200  *
201  * D = (S^{d}:S^{d})^{3/2} / ( (S:S)^{5/2} + (S^{d}:S^{d})^{5/4} )
202  *
203  * where S is the symmetric part of the velocity gradient
204  *
205  * S = 1/2 * (grad(u) + grad(u)^T) and S:S = S_ij * S_ij
206  *
207  * and S^{d} the traceless symmetric part of the square of the velocity
208  * gradient tensor
209  *
210  * S^{d} = 1/2 * (g^{2} + (g^{2})^T) - 1/3 * trace(g^{2}) * I
211  *
212  * with the square of the velocity gradient tensor
213  *
214  * g^{2} = grad(u) * grad(u)
215  *
216  * and the identity tensor I.
217  *
218  */
219  void
220  wale_model(scalar const & filter_width,
221  tensor const & velocity_gradient,
222  double const & C,
223  scalar & viscosity) const;
224 
225  /*
226  * Sigma model (Toda et al. 2010, Nicoud et al. 2011):
227  *
228  * nu_SGS = (C * filter_width)^{2} * D
229  *
230  * where the differential operator D is defined as
231  *
232  * D = s3 * (s1 - s2) * (s2 - s3) / s1^{2}
233  *
234  * where s1 >= s2 >= s3 >= 0 are the singular values of
235  * the velocity gradient tensor g = grad(u).
236  *
237  * The model constant is
238  *
239  * C = 1.35 (Nicoud et al. (2011)) ,
240  * C = 1.5 (Toda et al. (2010)) .
241  */
242  void
243  sigma_model(scalar const & filter_width,
244  tensor const & velocity_gradient,
245  double const & C,
246  scalar & viscosity) const;
