37/*
38 * This function calculates the penalty parameter of the interior
39 * penalty method for each cell.
40 */
41template<int dim, typename Number>
42void
43calculate_penalty_parameter(AlignedVector<VectorizedArray<Number>> & array_penalty_parameter,
44 MatrixFree<dim, Number> const & matrix_free,
45 unsigned int const dof_index = 0)
46{
47 unsigned int n_cells = matrix_free.n_cell_batches() + matrix_free.n_ghost_cell_batches();
48 array_penalty_parameter.resize(n_cells);
49
50 Mapping<dim> const & mapping = *matrix_free.get_mapping_info().mapping;
51 FiniteElement<dim> const & fe = matrix_free.get_dof_handler(dof_index).get_fe();
52 unsigned int const degree = fe.degree;
53
54 QGauss<dim> quadrature(degree + 1);
55 FEValues<dim> fe_values(mapping, fe, quadrature, update_JxW_values);
56
57 QGauss<dim - 1> face_quadrature(degree + 1);
58 FEFaceValues<dim> fe_face_values(mapping, fe, face_quadrature, update_JxW_values);
59
60 for(unsigned int i = 0; i < n_cells; ++i)
61 {
62 for(unsigned int v = 0; v < matrix_free.n_active_entries_per_cell_batch(i); ++v)
63 {
64 typename DoFHandler<dim>::cell_iterator cell = matrix_free.get_cell_iterator(i, v, dof_index);
65 fe_values.reinit(cell);
66
67 // calculate cell volume
68 Number volume = 0;
69 for(unsigned int q = 0; q < quadrature.size(); ++q)
70 {
71 volume += fe_values.JxW(q);
72 }
73
74 // calculate surface area
75 Number surface_area = 0;
76 for(const unsigned int f : cell->face_indices())
77 {
78 fe_face_values.reinit(cell, f);
79 Number const factor = (cell->at_boundary(f) && !cell->has_periodic_neighbor(f)) ? 1. : 0.5;
80 for(unsigned int q = 0; q < face_quadrature.size(); ++q)
81 {
82 surface_area += fe_face_values.JxW(q) * factor;
83 }
84 }
85
86 array_penalty_parameter[i][v] = surface_area / volume;
87 }
88 }
89}

91/*
92 * This function returns the penalty factor of the interior penalty method
93 * for quadrilateral/hexahedral elements for a given polynomial degree of
94 * the shape functions and a specified penalty factor (scaling factor).
95 */
96
97template<typename Number>
98Number
99get_penalty_factor(unsigned int const degree, Number const factor = 1.0)
100{
101 return factor * (degree + 1.0) * (degree + 1.0);
102}
