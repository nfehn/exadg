233  // description: see enum declaration
234  ProblemType problem_type;
235 
236  // set true in order to consider body forces
237  bool body_force;
238 
239  // are large deformations to be expected, than compute with non linear method
240  bool large_deformation;
241 
242  // For nonlinear problems with large deformations, it is important to specify whether
243  // the body forces are formulated with respect to the current deformed configuration
244  // or the reference configuration.
245  //
246  // Option 1: pull_back_body_force = false
247  // In this case, the body force is specified as force per undeformed volume. A typical
248  // use case are density-proportional forces such as the gravitational force. The body
249  // is then directly described in reference space, b_0 = rho_0 * g, and the pull-back to
250  // the reference configuration is deactivated.
251  //
252  // Option 2: pull_back_body_force = true
253  // The body force is specified as force per deformed volume, and the body force needs
254  // to be pulled-back according to b_0 = dv/dV * b, where the volume ratio dv/dV depends
255  // on the current state of deformation.
256  bool pull_back_body_force;
257 
258  // For nonlinear problems with large deformations, it is important to specify whether
259  // the traction Neumann boundary condition is formulated with respect to the current
260  // deformed configuration or the reference configuration. Both cases appear in practice,
261  // so it needs to be specified by the user which formulation is to be used.
262  //
263  // Option 1: pull_back_traction = false
264  // In this case, the traction is specified as a force per undeformed area, e.g., a
265  // force of fixed amount distributed uniformly over a surface of the body. The force per
266  // deformed area is an unknown. Hence, it is more natural to specify the traction in the
267  // reference configuration and deactivate the pull-back from the current to the reference
268  // configuration.
269  //
270  // Option 2: pull_back_traction = true
271  // The traction is known as a force per area of the deformed body. In this case, the
272  // traction needs to be pulled-back to the reference configuration, i.e., t_0 = da/dA * t,
273  // where the surface area ratio da/dA depends on the current state of deformation.
274  // A typical use case would be fluid-structure-interaction problems where the fluid
275  // stresses are applied as traction boundary conditions for the structure. Note that
276  // the direction of the traction vector does not change by this pull-back operation.
277  bool pull_back_traction;
