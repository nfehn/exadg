93  void
94  fill_grid_coordinates_vector(Mapping<dim> const & mapping,
95  VectorType & grid_coordinates,
96  DoFHandler<dim> const & dof_handler) const
97  {
98  if(grid_coordinates.size() != dof_handler.n_dofs())
99  {
100  IndexSet relevant_dofs_grid;
101  DoFTools::extract_locally_relevant_dofs(dof_handler, relevant_dofs_grid);
102  grid_coordinates.reinit(dof_handler.locally_owned_dofs(),
103  relevant_dofs_grid,
104  dof_handler.get_communicator());
105  }
106  else
107  {
108  grid_coordinates = 0;
109  }
110 
111  FiniteElement<dim> const & fe = dof_handler.get_fe();
112 
113  // Set up FEValues with base element and the Gauss-Lobatto quadrature to
114  // reduce setup cost, as we only use the geometry information (this means
115  // we need to call fe_values.reinit(cell) with Triangulation::cell_iterator
116  // rather than DoFHandler::cell_iterator).
117  FE_Nothing<dim> dummy_fe;
118  FEValues<dim> fe_values(mapping,
119  dummy_fe,
120  QGaussLobatto<dim>(fe.degree + 1),
121  update_quadrature_points);
122 
123  std::vector<types::global_dof_index> dof_indices(fe.dofs_per_cell);
124 
125  std::vector<std::array<unsigned int, dim>> component_to_system_index(
126  fe.base_element(0).dofs_per_cell);
127 
128  if(fe.dofs_per_vertex > 0) // FE_Q
129  {
130  for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
131  {
132  component_to_system_index
133  [hierarchic_to_lexicographic_numbering[fe.system_to_component_index(i).second]]
134  [fe.system_to_component_index(i).first] = i;
135  }
136  }
137  else // FE_DGQ
138  {
139  for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
140  {
141  component_to_system_index[fe.system_to_component_index(i).second]
142  [fe.system_to_component_index(i).first] = i;
143  }
144  }
145 
146  for(auto const & cell : dof_handler.active_cell_iterators())
147  {
148  if(!cell->is_artificial())
149  {
150  fe_values.reinit(typename Triangulation<dim>::cell_iterator(cell));
151  cell->get_dof_indices(dof_indices);
152  for(unsigned int i = 0; i < fe_values.n_quadrature_points; ++i)
153  {
154  Point<dim> const point = fe_values.quadrature_point(i);
155  for(unsigned int d = 0; d < dim; ++d)
156  {
157  if(grid_coordinates.get_partitioner()->in_local_range(
158  dof_indices[component_to_system_index[i][d]]))
159  {
160  grid_coordinates(dof_indices[component_to_system_index[i][d]]) = point[d];
161  }
162  }
163  }
164  }
165  }
166 
167  grid_coordinates.update_ghost_values();
168  }

182  void
183  initialize_mapping_q_cache(std::shared_ptr<Mapping<dim> const> mapping,
184  VectorType const & displacement_vector,
185  DoFHandler<dim> const & dof_handler)
186  {
187  AssertThrow(MultithreadInfo::n_threads() == 1, ExcNotImplemented());
188 
189  VectorType displacement_vector_ghosted;
190  if(dof_handler.n_dofs() > 0 && displacement_vector.size() == dof_handler.n_dofs())
191  {
192  IndexSet locally_relevant_dofs;
193  DoFTools::extract_locally_relevant_dofs(dof_handler, locally_relevant_dofs);
194  displacement_vector_ghosted.reinit(dof_handler.locally_owned_dofs(),
195  locally_relevant_dofs,
196  dof_handler.get_communicator());
197  displacement_vector_ghosted.copy_locally_owned_data_from(displacement_vector);
198  displacement_vector_ghosted.update_ghost_values();
199  }
200 
201  std::shared_ptr<FEValues<dim>> fe_values;
202 
203  FE_Nothing<dim> fe_nothing;
204  if(mapping.get() != 0)
205  {
206  fe_values = std::make_shared<FEValues<dim>>(*mapping,
207  fe_nothing,
208  QGaussLobatto<dim>(this->get_degree() + 1),
209  update_quadrature_points);
210  }
211 
212  // update mapping according to mesh deformation described by displacement vector
213  MappingQCache<dim>::initialize(
214  dof_handler.get_triangulation(),
215  [&](const typename Triangulation<dim>::cell_iterator & cell_tria) -> std::vector<Point<dim>> {
216  unsigned int const scalar_dofs_per_cell = Utilities::pow(this->get_degree() + 1, dim);
217 
218  std::vector<Point<dim>> grid_coordinates(scalar_dofs_per_cell);
219 
220  if(mapping.get() != 0)
221  {
222  fe_values->reinit(cell_tria);
223  // extract displacement and add to original position
224  for(unsigned int i = 0; i < scalar_dofs_per_cell; ++i)
225  {
226  grid_coordinates[i] =
227  fe_values->quadrature_point(this->hierarchic_to_lexicographic_numbering[i]);
228  }
229  }
230 
231  // if this function is called with an empty dof-vector, this indicates that the
232  // displacements are zero and the points do not have to be moved
233  if(dof_handler.n_dofs() > 0 && displacement_vector.size() > 0 && cell_tria->is_active() &&
234  !cell_tria->is_artificial())
235  {
236  FiniteElement<dim> const & fe = dof_handler.get_fe();
237  AssertThrow(fe.element_multiplicity(0) == dim,
238  ExcMessage("Expected finite element with dim components."));
239 
240  typename DoFHandler<dim>::cell_iterator cell(&cell_tria->get_triangulation(),
241  cell_tria->level(),
242  cell_tria->index(),
243  &dof_handler);
244 
245  std::vector<types::global_dof_index> dof_indices(fe.dofs_per_cell);
246  cell->get_dof_indices(dof_indices);
247 
248  for(unsigned int i = 0; i < dof_indices.size(); ++i)
249  {
250  std::pair<unsigned int, unsigned int> const id = fe.system_to_component_index(i);
251 
252  if(fe.dofs_per_vertex > 0) // FE_Q
253  {
254  grid_coordinates[id.second][id.first] += displacement_vector_ghosted(dof_indices[i]);
255  }
256  else // FE_DGQ
257  {
258  grid_coordinates[this->lexicographic_to_hierarchic_numbering[id.second]][id.first] +=
259  displacement_vector_ghosted(dof_indices[i]);
260  }
261  }
262  }
263 
264  return grid_coordinates;
265  });
266  }

269  std::vector<unsigned int> lexicographic_to_hierarchic_numbering;

279 template<int dim, typename Number>
280 void
281 initialize_multigrid(std::shared_ptr<MappingDoFVector<dim, Number>> mapping_multigrid,
282  std::shared_ptr<MappingQCache<dim> const> & mapping_q_cache,
283  Triangulation<dim> const & triangulation)
284 {
285  AssertThrow(MultithreadInfo::n_threads() == 1, ExcNotImplemented());
286 
287  typedef LinearAlgebra::distributed::Vector<Number> VectorType;
288 
289  // we have to project the solution onto all coarse levels of the triangulation
290  MGLevelObject<VectorType> grid_coordinates_all_levels, grid_coordinates_all_levels_ghosted;
291  unsigned int const n_levels = triangulation.n_global_levels();
292  grid_coordinates_all_levels.resize(0, n_levels - 1);
293  grid_coordinates_all_levels_ghosted.resize(0, n_levels - 1);
294 
295  FESystem<dim> fe(FE_Q<dim>(mapping_q_cache->get_degree()), dim);
296  DoFHandler<dim> dof_handler(triangulation);
297  dof_handler.distribute_dofs(fe);
298  dof_handler.distribute_mg_dofs();
299  VectorType grid_coordinates_fine_level;
300  mapping_multigrid->fill_grid_coordinates_vector(*mapping_q_cache,
301  grid_coordinates_fine_level,
302  dof_handler);
303 
304  MGTransferMatrixFree<dim, Number> transfer;
305  transfer.build(dof_handler);
306  transfer.interpolate_to_mg(dof_handler, grid_coordinates_all_levels, grid_coordinates_fine_level);
307 
308  for(unsigned int level = 0; level < n_levels; level++)
309  {
310  IndexSet relevant_dofs;
311  DoFTools::extract_locally_relevant_level_dofs(dof_handler, level, relevant_dofs);
312 
313  grid_coordinates_all_levels_ghosted[level].reinit(
314  dof_handler.locally_owned_mg_dofs(level),
315  relevant_dofs,
316  grid_coordinates_all_levels[level].get_mpi_communicator());
317 
318  grid_coordinates_all_levels_ghosted[level].copy_locally_owned_data_from(
319  grid_coordinates_all_levels[level]);
320 
321  grid_coordinates_all_levels_ghosted[level].update_ghost_values();
322  }
323 
324  AssertThrow(fe.element_multiplicity(0) == dim,
325  ExcMessage("Expected finite element with dim components."));
326 
327  // update mapping for all multigrid levels according to grid coordinates described by static
328  // mapping
329  mapping_multigrid->initialize(
330  dof_handler.get_triangulation(),
331  [&](const typename Triangulation<dim>::cell_iterator & cell_tria) -> std::vector<Point<dim>> {
332  unsigned int const level = cell_tria->level();
333 
334  typename DoFHandler<dim>::cell_iterator cell(&cell_tria->get_triangulation(),
335  level,
336  cell_tria->index(),
337  &dof_handler);
338 
339  unsigned int const scalar_dofs_per_cell = Utilities::pow(fe.degree + 1, dim);
340 
341  std::vector<Point<dim>> grid_coordinates(scalar_dofs_per_cell);
342 
343  if(cell->level_subdomain_id() != numbers::artificial_subdomain_id)
344  {
345  std::vector<types::global_dof_index> dof_indices(fe.dofs_per_cell);
346  cell->get_mg_dof_indices(dof_indices);
347 
348  for(unsigned int i = 0; i < dof_indices.size(); ++i)
349  {
350  std::pair<unsigned int, unsigned int> const id = fe.system_to_component_index(i);
351 
352  if(fe.dofs_per_vertex > 0) // FE_Q
353  {
354  grid_coordinates[id.second][id.first] =
355  grid_coordinates_all_levels_ghosted[level](dof_indices[i]);
356  }
357  else // FE_DGQ
358  {
359  grid_coordinates[mapping_multigrid->lexicographic_to_hierarchic_numbering[id.second]]
360  [id.first] = grid_coordinates_all_levels_ghosted[level](dof_indices[i]);
361  }
362  }
363  }
364 
365  return grid_coordinates;
366  });
367 }

373 template<int dim, typename Number>
374 void
375 initialize_multigrid(
376  std::vector<std::shared_ptr<MappingDoFVector<dim, Number>>> & coarse_grid_mappings,
377  std::shared_ptr<MappingQCache<dim> const> & mapping_q_cache,
378  std::vector<std::shared_ptr<Triangulation<dim> const>> & coarse_grid_triangulations)
379 {
380  typedef LinearAlgebra::distributed::Vector<Number> VectorType;
381 
382  FESystem<dim> fe(FE_Q<dim>(mapping_q_cache->get_degree()), dim);
383  unsigned int const n_h_levels = coarse_grid_triangulations.size();
384  std::vector<DoFHandler<dim>> coarse_grid_dof_handlers(n_h_levels);
385  std::vector<AffineConstraints<Number>> coarse_grid_constraints(n_h_levels);
386  for(unsigned int i = 0; i < n_h_levels; ++i)
387  {
388  coarse_grid_dof_handlers[i].reinit(*coarse_grid_triangulations[i]);
389  coarse_grid_dof_handlers[i].distribute_dofs(fe);
390  // constraints are irrelevant for interpolation
391  coarse_grid_constraints[i].close();
392  }
393  MGLevelObject<MGTwoLevelTransfer<dim, VectorType>> transfers(0, n_h_levels - 1);
394  for(unsigned int i = 1; i < n_h_levels; ++i)
395  {
396  transfers[i].reinit_geometric_transfer(coarse_grid_dof_handlers[i],
397  coarse_grid_dof_handlers[i - 1],
398  coarse_grid_constraints[i],
399  coarse_grid_constraints[i - 1]);
400  }
401 
402  // a function that initializes the dof-vector for a given level and dof_handler
403  const std::function<void(const unsigned int, VectorType &)> initialize_dof_vector =
404  [&](const unsigned int h_level, VectorType & vector) {
405  IndexSet locally_relevant_dofs;
406  DoFTools::extract_locally_relevant_dofs(coarse_grid_dof_handlers[h_level],
407  locally_relevant_dofs);
408  vector.reinit(coarse_grid_dof_handlers[h_level].locally_owned_dofs(),
409  locally_relevant_dofs,
410  coarse_grid_dof_handlers[h_level].get_communicator());
411  };
412 
413  dealii::MGTransferGlobalCoarsening<dim, VectorType> mg_transfer_global_coarsening(
414  transfers, initialize_dof_vector);
415 
416  MGLevelObject<VectorType> coarse_grid_coordinates(0, n_h_levels - 1);
417 
418  coarse_grid_mappings.resize(n_h_levels);
419  coarse_grid_mappings[n_h_levels - 1] =
420  std::make_shared<MappingDoFVector<dim, Number>>(mapping_q_cache->get_degree());
421 
422  // get dof-vector with grid coordinates from the finest h-level
423  coarse_grid_mappings[n_h_levels - 1]->fill_grid_coordinates_vector(
424  *mapping_q_cache,
425  coarse_grid_coordinates[n_h_levels - 1],
426  coarse_grid_dof_handlers[n_h_levels - 1]);
427 
428  // transfer grid coordinates to coarser h-levels
429  // the DoFHandler object will not be used for global coarsening
430  DoFHandler<dim> dof_handler_dummy;
431  VectorType vector_copy(coarse_grid_coordinates[n_h_levels - 1]);
432  mg_transfer_global_coarsening.interpolate_to_mg(dof_handler_dummy,
433  coarse_grid_coordinates,
434  vector_copy);
435 
436  // initialize mapping for all h-levels using the dof-vectors with grid coordinates
437  for(unsigned int h_level = 0; h_level < n_h_levels; ++h_level)
438  {
439  coarse_grid_mappings[h_level] =
440  std::make_shared<MappingDoFVector<dim, Number>>(mapping_q_cache->get_degree());
441 
442  // coarse_grid_coordinates describes absolute coordinates -> use an uninitialized mapping
443  std::shared_ptr<Mapping<dim> const> mapping_dummy;
444  coarse_grid_mappings[h_level]->initialize_mapping_q_cache(mapping_dummy,
445  coarse_grid_coordinates[h_level],
446  coarse_grid_dof_handlers[h_level]);
447  }
448 }
