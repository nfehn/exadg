101template<int dim, int data_dim, typename VectorizedArrayType>
102void
103DoFCoupling<dim, data_dim, VectorizedArrayType>::define_coupling_mesh()
104{
105 Assert(this->mesh_id != -1, dealii::ExcNotInitialized());
106
107 // In order to avoid that we define the surface multiple times when reader
108 // and writer refer to the same object
109 if(coupling_nodes_ids.size() > 0)
110 return;
111
112 // Get and sort the global dof indices
113 auto const get_component_dofs = [&](const int component) {
114 // Get a component mask of the vector component
115 dealii::ComponentMask component_mask(data_dim, false);
116 component_mask.set(component, true);
117
118 // Get the global DoF indices of the component
119 // Compute the intersection with locally owned dofs
120 // TODO: This is super inefficient, have a look at the
121 // dof_handler.n_boundary_dofs implementation for a proper version
122 const dealii::IndexSet indices =
123 (dealii::DoFTools::extract_boundary_dofs(this->matrix_free->get_dof_handler(mf_dof_index),
124 component_mask,
125 std::set<dealii::types::boundary_id>{
126 this->dealii_boundary_surface_id}) &
127 this->matrix_free->get_dof_handler(mf_dof_index).locally_owned_dofs());
128
129 Assert(indices.n_elements() * data_dim ==
130 this->matrix_free->get_dof_handler(mf_dof_index)
131 .n_boundary_dofs(
132 std::set<dealii::types::boundary_id>{this->dealii_boundary_surface_id}),
133 dealii::ExcInternalError());
134 // Resize the global dof index conatiner in case we call this lambda for
135 // the first time
136 if(component == 0)
137 global_indices.resize(indices.n_elements());
138 // fill the first array entry with the respective component
139 dealii::types::global_dof_index iterator = 0;
140 for(const auto dof : indices)
141 {
142 global_indices[iterator][component] = dof;
143 ++iterator;
144 }
145 };
146
147 // Fill the indices object this class works on
148 for(int d = 0; d < data_dim; ++d)
149 get_component_dofs(d);
150 // Compute the coordinates of the indices (only one component required here)
151 // We select the zeroth component
152 std::map<dealii::types::global_dof_index, dealii::Point<dim>> support_points;
153 dealii::ComponentMask component_mask(data_dim, false);
154 component_mask.set(0, true);
155
156 dealii::DoFTools::map_boundary_dofs_to_support_points(
157 *(this->matrix_free->get_mapping_info().mapping),
158 this->matrix_free->get_dof_handler(mf_dof_index),
159 support_points,
160 component_mask,
161 this->dealii_boundary_surface_id);
162
163
164 // Set size of the preCICE ID vector
165 coupling_nodes_ids.reserve(global_indices.size());
166 std::array<double, dim> nodes_position;
167 for(std::size_t i = 0; i < global_indices.size(); ++i)
168 {
169 // Get index of the zeroth component
170 const auto element = global_indices[i][0];
171 for(int d = 0; d < dim; ++d)
172 nodes_position[d] = support_points[element][d];
173
174 // pass node coordinates to precice
175 const int precice_id = this->precice->setMeshVertex(this->mesh_id, nodes_position.data());
176 coupling_nodes_ids.emplace_back(precice_id);
177 }
178
179 if(this->read_data_map.size() > 0)
180 this->print_info(true, this->precice->getMeshVertexSize(this->mesh_id));
181 if(this->write_data_map.size() > 0)
182 this->print_info(false, this->precice->getMeshVertexSize(this->mesh_id));
183}

187template<int dim, int data_dim, typename VectorizedArrayType>
188void
189DoFCoupling<dim, data_dim, VectorizedArrayType>::write_data(
190 const dealii::LinearAlgebra::distributed::Vector<double> & data_vector,
191 const std::string & data_name)
192{
193 const int write_data_id = this->write_data_map.at(data_name);
194 Assert(write_data_id != -1, dealii::ExcNotInitialized());
195 Assert(coupling_nodes_ids.size() > 0, dealii::ExcNotInitialized());
196
197 std::array<double, data_dim> write_data;
198 for(std::size_t i = 0; i < global_indices.size(); ++i)
199 {
200 // Extract relevant elements from global vector
201 for(unsigned int d = 0; d < data_dim; ++d)
202 {
203 const auto element = global_indices[i][d];
204 write_data[d] = data_vector[element];
205 }
206
207 // and pass them to preCICE
208 if constexpr(data_dim > 1)
209 {
210 this->precice->writeVectorData(write_data_id, coupling_nodes_ids[i], write_data.data());
211 }
212 else
213 {
214 this->precice->writeScalarData(write_data_id, coupling_nodes_ids[i], write_data[0]);
215 }
216 }
217}
